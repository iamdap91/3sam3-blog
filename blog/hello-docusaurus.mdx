### Show Line Number

```js showLineNumbers
export const formatDate = (date: string) => {
    const regex = /^(\d{4})(\d{2})(\d{2})$/;
    if(!date.match(regex)) {
        throw new Error('날짜 포맷에 맞지 않습니다.')
    }

    return date.replace(regex, '$1-$2-$3');
};
```
